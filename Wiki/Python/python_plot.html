import matplotlib.pyplot as plt
import numpy as np
fig = plt.figure()
ax = plt.axes() 

x = np.linspace(0, 10, 1000)
ax.plot(x, np.sin(x));

# For short, you can use the following codes:
ax.plot(x, x + 4, linestyle='-')  # solid
ax.plot(x, x + 5, linestyle='--') # dashed
ax.plot(x, x + 6, linestyle='-.') # dashdot

ax.plot(x, x + 7, linestyle=':', label='dotted');  # dotted # Labeling Plots
plt.legend(); # add Labeling 

# set labels
plt.xlabel('time (s)')
plt.ylabel('voltage (mV)')

# set title
plt.title('Simple Plot')
plt.grid(True)

plt.savefig("pyton_plot_test.png")

import matplotlib.pyplot as plt
from IPython.display import display
import numpy as np
from ipywidgets import interactive

def plot_sine(frequency=1.0, amplitude=1.0):
    plt.ylim(-1.0, 1.0);
    x = np.linspace(0, 10, 1000)
    plt.plot(x, amplitude*np.sin(x*frequency));

v=interactive(plot_sine, frequency=(0.5, 10.0), amplitude=(0.0, 1.0))
